> with(combinat)


Bai 3: Cho n va r la nhung so nguyen duong. Hay tinh he so cua x^n cua ham sinh

a) 1/(1-x)(1-2x)..(1-rx)

> HeSoA := proc (r, n) 
     local f := product(1/(-i*x+1), i = 1 .. r); 
     local G := series(f, x, n); 
     coeff(G, x, n-1) ;
    end proc;

> HeSoA(5, 6);
  

b) 1/(1-x)(1-x^2)...(1-x^r)

> HeSoB := proc (r, n) 
      local f := product(1/(1-x^i), i = 1 .. r); 
      local G := series(f, x, n); 
      coeff(G, x, n-1); 
    end proc;


> HeSoB(5, 6);
